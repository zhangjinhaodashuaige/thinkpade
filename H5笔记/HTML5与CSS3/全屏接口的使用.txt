    full.onclick=function(){

注释//            火狐浏览器
            div.mozRequestFullScreen();
        }
        else if(div.requestFullScreen){

            div.requestFullScreen();
        }
        else if(div.webkitRequestFullScreen){
            div.webkitRequestFullScreen();
        }
注释//        微软
        else  if(div.msRequestFullscreen){
            div.msRequestFullscreen();
        }
    }
注释//cancelFullScreen退出全屏事件，因为不同的浏览内核不一样的，所以在事件前面加的浏览器前缀不一样
注释// 所以要进行能力测试
注释//
    cancelFull.onclick=function(){
注释//      因为现在是文档全屏所以要让文档退出全屏
        if(document.mozCancelFullScreen){
            document.mozCancelFullScreen();
        }
        else if(document.cancelFullScreen){
            document.cancelFullScreen();
        }
        else if(document.webkitCancelFullScreen){
            document.webkitCancelFullScreen();
        }
        else  if(document.msExitFullscreen){
            document.msExitFullscreen();

注释//所以要进行浏览器能力测试
    isFull.onclick=function(){
注释     /*判断  1.判断当前文档   2.能力测试*/
        if(document.fullScreenElement || document.mozFullScreenElement || document.webkitFullscreenElement  || document.msFullscreenElement  ||document.oFullscreenElement){
